-- CreateSection proxy: captures create* returns + FindFirstChild adapter (updateText->Set)
local _ctrls = {}
local _origCreateSection = Setup.CreateSection
local function _adapt(obj)
    if not obj then return nil end
    return setmetatable({}, { __index = function(_, k)
        if k == "updateText" then return function(_, t) if obj.Set then obj:Set(t) elseif obj.updateText then obj:updateText(t) end end end
        if k == "Set" then return function(_, ...) if obj.Set then obj:Set(...) end end end
        if k == "updateList" then return function(_, l) if obj.updateList then obj:updateList(l) end end end
        local v = obj[k]
        if type(v) == "function" then return function(_, ...) return obj[k](obj, ...) end end
        return v
    end})
end
function Setup:CreateSection(name)
    local sec = _origCreateSection(self, name)
    local proxy = setmetatable({}, {
        __index = function(_, k)
            if k == "FindFirstChild" then return function(_, tag) return _adapt(_ctrls[tag]) end end
            return sec[k]
        end,
    })
    for _, m in ipairs({ "createLabel", "createDropdown", "createToggle", "createSlider", "createButton", "createInputBox" }) do
        proxy[m] = function(_, cfg)
            local ok, obj = pcall(sec[m], sec, cfg)
            if cfg and cfg.flagName and obj then _ctrls[cfg.flagName] = obj end
            return obj
        end
    end
    return proxy
end

-- interval helper
local function interval(tag, flag, delayTime, callback)
    Library:CleanupConnectionsByTag(tag)
    delayTime = math.max(tonumber(delayTime) or 0.1, 0.05)
    if not Library.Flags[flag] then return end
    local last = 0
    local running = false
    local conn = RunService.Heartbeat:Connect(function()
        if not Library.Flags[flag] then Library:CleanupConnectionsByTag(tag) return end
        local now = os.clock()
        if running or now - last < delayTime then return end
        last = now
        running = true
        task.spawn(function()
            local ok, err = pcall(callback)
            if not ok then warn("[GAG:" .. tostring(tag) .. "]", err) end
            task.wait()
            running = false
        end)
    end)
    Library:TrackConnection(conn, tag)
end

end

-- netFire helper (walks Net by dotted path, pcall-guarded)
local function netFire(path, ...)
    if not Net then return end
    local node = Net
    for p in string.gmatch(path, "[^%.]+") do node = node and node[p] end
    if type(node) == "table" then
        if type(node.Fire) == "function" then pcall(node.Fire, node, ...)
        elseif type(node.Invoke) == "function" then pcall(node.Invoke, node, ...) end
    end
end

local function teleport(pos)
    local r = getHRP() 
    if not (r and pos) then return end 
    local target = pos + Vector3.new(0, 3, 0) 

    if Library.Flags["UseVersusAI"] then
        -- Wait for VersusAI backend integration (Learned Nodes)
        -- Fallback to smooth interpolation
        for _ = 1, 90 do
            local cur = r.Position
            local delta = target - cur
            if delta.Magnitude <= 10 then r.CFrame = CFrame.new(target) break end
            r.CFrame = CFrame.new(cur + delta.Unit * 10)
            RunService.Heartbeat:Wait()
        end
        return
    end

    for _ = 1, 60 do 
        local cur = r.Position 
        local delta = target - cur 
        if delta.Magnitude <= HOP then r.CFrame = CFrame.new(target) break end 
        r.CFrame = CFrame.new(cur + delta.Unit * HOP) 
        RunService.Heartbeat:Wait() 
    end 
end

-- ripe check
local function isRipe(m)
    local age = tonumber(m:GetAttribute("Age")) local mx = tonumber(m:GetAttribute("MaxAge"))
    if age and mx then return age >= mx - 0.001 end
    for _, d in ipairs(m:GetDescendants()) do if d:IsA("ProximityPrompt") and CollectionService:HasTag(d, "HarvestPrompt") then return true end end
    return false
end

-- get ripe crops with full metadata for filtering
local function getRipeCrops()
    local out = {}
    local plot = myPlot() if not plot then return out end
    local plants = plot:FindFirstChild("Plants") if not plants then return out end
    for _, plant in ipairs(plants:GetChildren()) do
        local fr = plant:FindFirstChild("Fruits")
        local fruits = fr and fr:GetChildren() or {}
        if #fruits > 0 then
            for _, m in ipairs(fruits) do
                if isRipe(m) then
                    local pid = m:GetAttribute("PlantId")
                    if pid then
                        out[#out + 1] = {
                            model = m, plantId = pid, fruitId = m:GetAttribute("FruitId") or "",
                            mutation = m:GetAttribute("Mutation"),
                            crop = m:GetAttribute("CorePartName") or m:GetAttribute("SeedName") or "",
                            weight = tonumber(m:GetAttribute("SizeMultiplier") or m:GetAttribute("SizeMulti")) or 1,
                            rarity = m:GetAttribute("Rarity"),
                            value = (SeedBaseValue[m:GetAttribute("CorePartName") or m:GetAttribute("SeedName")] or 0) * sizeMul(m:GetAttribute("SizeMulti") or 1) * (m:GetAttribute("Mutation") and MUT_BONUS or 1),
                        }
                    end
                end
            end
        elseif isRipe(plant) then
            local pid = plant:GetAttribute("PlantId")
            if pid then
                out[#out + 1] = {
                    model = plant, plantId = pid, fruitId = "",
                    mutation = plant:GetAttribute("Mutation"),
                    crop = plant:GetAttribute("SeedName") or plant:GetAttribute("CorePartName") or "",
                    weight = 1, rarity = plant:GetAttribute("Rarity"), value = 0,
                }
            end
        end
    end
    return out
end

-- filter quintet matcher: checks fruit against rarity/mutation/threshold filters
-- any of fType/fRarity/fMutation/fThreshMode/fThreshold may be nil (skip that check).
local function matchesFilter(entry, fType, fRarity, fMutation, fThreshMode, fThreshold)
    if fType then
        local ft = firstValue(Library.Flags[fType] or {})
        if ft and ft ~= "" and ft ~= "Any" and entry.crop and entry.crop:lower() ~= tostring(ft):lower() then return false end
    end
    if fRarity then
        local rar = firstValue(Library.Flags[fRarity] or {})
        if rar and rar ~= "Any" then
            local entryRar = RARITY_ORDER[entry.rarity or "Common"] or 1
            local minRar = RARITY_ORDER[rar] or 1
            if entryRar < minRar then return false end
        end
    end
    if fMutation then
        local mut = firstValue(Library.Flags[fMutation] or {})
        if mut == "Mutated Only" and not entry.mutation then return false end
        if mut == "Non-Mutated" or mut == "Non-Mutated Only" then if entry.mutation then return false end end
    end
    if fThreshMode and fThreshold then
        local tm = firstValue(Library.Flags[fThreshMode] or {})
        local th = tonumber(Library.Flags[fThreshold]) or 0
        if tm and tm ~= "Disabled" and th > 0 then
            if tm == "Weight" and (entry.weight or 0) < th then return false end
            if tm == "Value" and (entry.value or 0) < th then return false end
        end
    end
    return true
end

-- soil/plant slot helpers
local function getSoilAreas(plot)
    local areas = {}
    for _, p in ipairs(CollectionService:GetTagged("PlantArea")) do
        if p:IsA("BasePart") and p:IsDescendantOf(plot) and p.Size.X * p.Size.Z > 400 then areas[#areas + 1] = p end
    end
    if #areas == 0 then local ref = plot:FindFirstChild("PlotSizeReference") if ref then areas = { ref } end end
    return areas
end
local PLANT_PATTERNS = { "Fill", "Checkerboard", "Rows", "Columns", "Diagonal", "Spaced" }
local function getPlantSlots(plot, pattern)
    pattern = pattern or "Fill" local step = 6 local seen, list = {}, {}
    for _, area in ipairs(getSoilAreas(plot)) do
        local cf = area.CFrame local sz = area.Size local topY = area.Position.Y + sz.Y / 2 + 0.3
        local hx = sz.X / 2 - 3 local hz = sz.Z / 2 - 3
        local nx = math.floor((2 * hx) / step) local nz = math.floor((2 * hz) / step)
        for ix = 0, nx do for iz = 0, nz do
            local w = (cf * CFrame.new(-hx + ix * step, 0, -hz + iz * step)).Position
            local gx, gz = math.floor(w.X / step + 0.5), math.floor(w.Z / step + 0.5)
            local keep = true
            if pattern == "Checkerboard" then keep = (gx + gz) % 2 == 0
            elseif pattern == "Rows" then keep = gz % 2 == 0
            elseif pattern == "Columns" then keep = gx % 2 == 0
            elseif pattern == "Diagonal" then keep = (gx - gz) % 3 == 0
            elseif pattern == "Spaced" then keep = gx % 2 == 0 and gz % 2 == 0 end
            if keep then local key = math.floor(w.X / 4 + 0.5) .. "," .. math.floor(w.Z / 4 + 0.5)
                if not seen[key] then seen[key] = true list[#list + 1] = Vector3.new(w.X, topY, w.Z) end end
        end end
    end
    return list
end
local function getOpenSlots(plot, pattern)
    local grid = getPlantSlots(plot, pattern) local plants = plot:FindFirstChild("Plants") local occ = {}
    if plants then for _, pl in ipairs(plants:GetChildren()) do local ok, pv = pcall(function() return pl:GetPivot().Position end) if ok then occ[#occ + 1] = pv end end end
    local free = {}
    for _, pos in ipairs(grid) do local clear = true
        for _, o in ipairs(occ) do if (Vector3.new(o.X, 0, o.Z) - Vector3.new(pos.X, 0, pos.Z)).Magnitude < 6 then clear = false break end end
        if clear then free[#free + 1] = pos end
    end
    return free
end

local function holdSeedPrompts(pos)
    local map = Workspace:FindFirstChild("Map")
    for _, cont in ipairs({ map and map:FindFirstChild("SeedPackSpawnServerLocations"), map and map:FindFirstChild("SeedPackSpawnClient"), Workspace:FindFirstChild("Temporary") }) do
        if cont then for _, d in ipairs(cont:GetDescendants()) do
            if d:IsA("ProximityPrompt") then
                local p = d.Parent local ok, pp = pcall(function() return p.Position end)
                if (not ok) or (pp - pos).Magnitude <= 35 then
                    pcall(function()
                        local hold = tonumber(d.HoldDuration) or 0
                        if fireproximityprompt then fireproximityprompt(d, math.max(hold, 0))
                        else d:InputHoldBegin(); task.wait(hold + 0.1); d:InputHoldEnd() end
                    end)
                end
            end
        end end
    end
end
local function locPos(loc)
    if loc:IsA("BasePart") then return loc.Position end
    local ok, cf = pcall(function() return loc:GetPivot() end) if ok then return cf.Position end
    local bp = loc:IsA("BasePart") and loc or loc:FindFirstChildWhichIsA("BasePart", true) return bp and bp.Position or nil
end
local function grabPackRobust(loc)
    local landed = false
    for _ = 1, 90 do
        if not (loc and loc.Parent) then break end
        local pos = locPos(loc) if not pos then break end
        local r = getHRP()
        if (not landed) or (r and (r.Position - pos).Magnitude > 6) then teleport(pos); landed = true end
        for _, d in ipairs(loc:GetDescendants()) do if d:IsA("ProximityPrompt") then pcall(function() local h = tonumber(d.HoldDuration) or 0 if fireproximityprompt then fireproximityprompt(d, math.max(h, 0)) else d:InputHoldBegin(); task.wait(h + 0.1); d:InputHoldEnd() end end) end end
        holdSeedPrompts(pos)
        local part = loc:IsA("BasePart") and loc or loc:FindFirstChildWhichIsA("BasePart", true)
        if firetouchinterest and part and r then pcall(function() firetouchinterest(r, part, 0); firetouchinterest(r, part, 1) end) end
        task.wait(0.12)
    end
end

local function captureSnapshot(name)
    local plot = gardenNearPlayer() if not plot then return false, "stand near a garden first" end
    local ref = plot:FindFirstChild("PlotSizeReference") local center = ref and ref.Position or Vector3.zero
    local snap = { seeds = {}, buildings = {}, owner = plot:GetAttribute("OwnerUserId") }
    local plants = plot:FindFirstChild("Plants")
    if plants then for _, pl in ipairs(plants:GetChildren()) do local s = pl:GetAttribute("SeedName") or pl:GetAttribute("CorePartName") if s then snap.seeds[s] = (snap.seeds[s] or 0) + 1 end end end
    for _, fname in ipairs(BUILD_FOLDERS) do local f = plot:FindFirstChild(fname)
        if f then for _, b in ipairs(f:GetChildren()) do local ok, piv = pcall(function() return b:GetPivot().Position end)

                snap.buildings[#snap.buildings + 1] = { kind = tostring(kind), folder = fname, rx = piv.X - center.X, ry = piv.Y - center.Y, rz = piv.Z - center.Z, rot = 0 } end
        end end
    end
    Snapshots[name] = snap saveSnapshots()
    local nSeeds = 0 for _ in pairs(snap.seeds) do nSeeds = nSeeds + 1 end
    return true, string.format("captured %d seed types, %d buildings", nSeeds, #snap.buildings)

    for _, pl in ipairs(plants:GetChildren()) do
        local pid = pl:GetAttribute("PlantId")
        local crop = pl:GetAttribute("SeedName") or pl:GetAttribute("CorePartName")
        if pid and ((not matchFn) or matchFn(crop)) then
            local ok, pos = pcall(function() return pl:GetPivot().Position end)

            netFire("Shovel.UseShovel", pid, "", sa, sh)
            n = n + 1 task.wait(0.05)
        end
    end
    return n
end

-- Inventory worth (sum of fruit weights * base values)
-- =======================================================
local function inventoryValue()
    local d = getData() if not d then return 0, 0 end
    local inv = d.Inventory and (d.Inventory.Fruits or d.Inventory.HarvestedFruits or d.Inventory.Crops)
    if not inv then return 0, 0 end
    local total, count = 0, 0
    for _, entry in pairs(inv) do
        local nm = (type(entry) == "table" and (entry.Name or entry.Type or entry.FruitType)) or tostring(entry)
        local wt = (type(entry) == "table" and (entry.Weight or entry.Size)) or 1
        local base = SeedBaseValue and SeedBaseValue[nm] or 0
        total = total + base * sizeMul(wt)
        count = count + 1
    end
    return total, count
end

-- feature functions
local function doHarvest(respectFilter)
    local plot = myPlot() local ref = plot and plot:FindFirstChild("PlotSizeReference") local r = getHRP()
    if not Library.Flags["collectNoTp"] then
        if ref and r and (Vector3.new(r.Position.X, 0, r.Position.Z) - Vector3.new(ref.Position.X, 0, ref.Position.Z)).Magnitude > 16 then teleport(ref.Position) task.wait(0.12) end
    end
    local targets = getRipeCrops() local n = 0
    -- mode: autoCollectAll overrides everything; autoCollectBest = only best; enableFilters = apply quintet
    local mode = "All"
    if Library.Flags["autoCollectAll"] then mode = "All"
    elseif Library.Flags["autoCollectBest"] then mode = "Best"
    elseif Library.Flags["enableFilters"] or respectFilter then mode = firstValue(Library.Flags["collectFilter"] or {}) or "All" end
    for _, entry in ipairs(targets) do
        local skip = false
        if mode == "Filtered" then
            if not matchesFilter(entry, "collectFruit", "collectRarity", "collectMutation", "collectThreshMode", "collectThreshold") then skip = true end
        elseif mode == "Best" then
            local bestV = 0 for _, e2 in ipairs(targets) do if (e2.value or 0) > bestV then bestV = e2.value or 0 end end
            if (entry.value or 0) < bestV then skip = true end
        end
        if Library.Flags["mutatedOnly"] and not entry.mutation then skip = true end
        if Library.Flags["stopOnFull"] and isInventoryFull() then break end
        if not skip then netFire("Garden.CollectFruit", entry.plantId, entry.fruitId) n = n + 1 task.wait(tonumber(Library.Flags["collectDelay"]) or 0.05) end
    end
    return n
end

local function doPlant()
    local plot = myPlot() if not plot then return end
    
    -- Teleport to garden so server distance checks pass
    local ref = plot:FindFirstChild("PlotSizeReference") local r = getHRP()
    if not Library.Flags["plantNoTp"] then
        if ref and r and (Vector3.new(r.Position.X, 0, r.Position.Z) - Vector3.new(ref.Position.X, 0, ref.Position.Z)).Magnitude > 16 then teleport(ref.Position) task.wait(0.12) end
    end

    local d = getData() local seeds = d and d.Inventory and d.Inventory.Seeds if not seeds then return end
    
    local seedFilter = Library.Flags["plantSeeds"]
    local hasFilter = false
    if type(seedFilter) == "table" and next(seedFilter) ~= nil then hasFilter = true
    elseif type(seedFilter) == "string" and seedFilter ~= "" then hasFilter = true end

    local toPlant = {}
    if Library.Flags["smartReplant"] then
        local best = getBestSeed()
        if best and seeds[best] and seeds[best] > 0 then
            for _ = 1, math.min(seeds[best], tonumber(Library.Flags["maxPerCycle"]) or 40) do toPlant[#toPlant + 1] = best end
        end
    elseif Library.Flags["autoPlantAll"] then
        for name, count in pairs(seeds) do if (count or 0) > 0 then for _ = 1, math.min(count or 0, 40) do toPlant[#toPlant + 1] = name end end end
    else
        for name, count in pairs(seeds) do
            if hasFilter then 
                local match = false 
                if type(seedFilter) == "table" then
                    for k, v in pairs(seedFilter) do 
                        if v == true or type(k) == "number" then
                            local f = type(k) == "number" and v or k
                            if type(f) == "string" and name:lower() == f:lower() then match = true break end 
                        end
                    end 
                elseif type(seedFilter) == "string" then
                    if name:lower() == seedFilter:lower() then match = true end
                end
                if not match then continue end 
            end
            for _ = 1, math.min(count or 0, tonumber(Library.Flags["maxPerCycle"]) or 40) do toPlant[#toPlant + 1] = name end
        end
    end
    if #toPlant == 0 then return end
    
    local free = getOpenSlots(plot, firstValue(Library.Flags["plantPattern"]) or "Fill") if #free == 0 then return end
    local cap = math.min(#free, #toPlant, tonumber(Library.Flags["maxPerCycle"]) or 40)
    local delay = tonumber(Library.Flags["plantDelay"]) or 0.1
    for i = 1, cap do netFire("Plant.PlantSeed", free[i], toPlant[i], plot) task.wait(delay) end
end

local function doSteal()
    -- Stealing is server-gated to nighttime — no client-side night check needed.
    local targets = getStealTargets() if #targets == 0 then return end
    local mode = firstValue(Library.Flags["stealFilter"] or {}) or "All"
    if Library.Flags["autoStealBest"] then
        table.sort(targets, function(a, b) return (a.value or 0) > (b.value or 0) end)
    end
    local home = getHRP() and getHRP().Position
    local lastPos
    for _, t in ipairs(targets) do
        if mode == "Filtered" then
            if not matchesFilter(t, "stealFruit", "stealRarity", "stealMutation", nil, nil) then continue end
        end
        local pos = (t.model and t.model.Parent) and t.model:GetPivot().Position or nil
        if pos and (not lastPos or (pos - lastPos).Magnitude > 12) then teleport(pos); lastPos = pos end
        -- verified: BeginSteal(userId, plantId, fruitId) — correct arg order
        netFire("Steal.BeginSteal", t.userId, t.plantId, t.fruitId)
        task.wait(tonumber(Library.Flags["stealDelay"]) or 0.3)
        -- verified: CompleteSteal() — 0 args. Fire multiple times to grab multi-fruit crops
        local mult = math.max(1, tonumber(Library.Flags["stealMult"]) or 1)
        for _ = 1, mult do netFire("Steal.CompleteSteal") task.wait(0.05) end
        if Library.Flags["autoStealBest"] then break end
    end
    if Library.Flags["stealReturn"] and home then teleport(home - Vector3.new(0, 3, 0)) end
end

local function doSprinkler()
    local d = getData() if not (d and d.Inventory and d.Inventory.Sprinklers) then return end
    local plot = myPlot() if not plot then return end

    local ref = plot:FindFirstChild("PlotSizeReference") local r = getHRP()
    if not Library.Flags["sprinklerNoTp"] then
        if ref and r and (Vector3.new(r.Position.X, 0, r.Position.Z) - Vector3.new(ref.Position.X, 0, ref.Position.Z)).Magnitude > 16 then teleport(ref.Position) task.wait(0.12) end
    end

    -- soft cap: honour server-side sprinkler cap by attribute, else fall back to plot expansion + 2
    local existing = plot:FindFirstChild("Sprinklers") local count = existing and #existing:GetChildren() or 0
    local maxCap = tonumber(plot:GetAttribute("MaxSprinklers"))
    if not maxCap then maxCap = 4 + (tonumber(plot:GetAttribute("GardenExpansion")) or 0) * 2 end
    if count >= maxCap then return end

    local pickedName = firstValue(Library.Flags["sprinklerSelect"] or {})
    local function firstOwned()
        for sname, sval in pairs(d.Inventory.Sprinklers) do
            local cnt = (type(sval) == "table" and (sval.Count or sval.Amount)) or (tonumber(sval) or 0)
            if type(sname) == "string" and cnt > 0 then return sname end
        end
    end

    -- position mode
    local posMode = firstValue(Library.Flags["sprinklerPos"] or {}) or "Random"
    local savedPos = Library.Flags["savedSprinklerPos"]
    local spacing = tonumber(Library.Flags["sprinklerSpacing"]) or 8
    local areas = getSoilAreas(plot) if #areas == 0 then return end

    local function pickPos()
        if posMode == "Saved Position" and typeof(savedPos) == "Vector3" then return savedPos + Vector3.new(0, 2, 0) end
        if posMode == "Player Position" then local h = getHRP() if h then return h.Position + Vector3.new(0, 2, 0) end end
        if posMode == "Near Fruit" then
            local ripe = getRipeCrops()
            if #ripe > 0 then
                local m = ripe[math.random(1, #ripe)].model
                local ok, p = pcall(function() return m:GetPivot().Position end)
                if ok then return p + Vector3.new(math.random(-spacing, spacing), 2, math.random(-spacing, spacing)) end
            end
        end
        -- Random inside the first soil area
        local a = areas[1] local hx, hz = a.Size.X/2 - 3, a.Size.Z/2 - 3
        return (a.CFrame * CFrame.new(math.random(-hx, hx), 0, math.random(-hz, hz))).Position + Vector3.new(0, 2, 0)
    end

    -- If "Auto Place All Sprinklers" is on, place until we run out of inventory or hit the cap
    local placeAll = Library.Flags["autoSprinklerAll"] == true

local function doMailboxSend()
    local username = Library.Flags["mbUsername"] or "" if username == "" then return end

    local d = getData() if not d then return end

    local ok, result = pcall(function()

        local lookup = Net.Mailbox and Net.Mailbox.LookupPlayer
        if not lookup then return nil end
        return lookup:Invoke(username)
    end)
    if not (ok and result) then return end
    local userId = type(result) == "table" and result[1] or result
    if not userId then return end
    -- send items based on toggles
    if Library.Flags["autoSendSeed"] and d.Inventory and d.Inventory.Seeds then

local _espPool = {}

local function doESP()
    clearESP()
    local function esp(model, col)
        if not (model and model.Parent) then return end
        local h = Instance.new("Highlight") h.Adornee = model h.FillColor = col h.FillTransparency = 0.4
        h.OutlineColor = col h.DepthMode = Enum.HighlightDepthMode.AlwaysOnTop h.Parent = CoreGui _espPool[#_espPool + 1] = h

                if (not target) or (crop and crop:lower() == target:lower()) then
                    if matchesFilter(entry, nil, "espFruitRarity", "espFruitMutation", nil, nil) then esp(m, Color3.fromRGB(0, 200, 255)) end
                end
            end
        end

            if (not target) or (nm and nm:lower() == target:lower()) then esp(pet, Color3.fromRGB(255, 100, 200)) end
        end end
    end
end

-- pack grab
local _grabbingPacks = {}
local function doPackGrab()
    local spawns = Workspace:FindFirstChild("Map") spawns = spawns and spawns:FindFirstChild("SeedPackSpawnServerLocations")
    if not spawns then return end
    for _, loc in ipairs(spawns:GetChildren()) do
        if loc.Parent and not _grabbingPacks[loc] then
            local has = loc:GetAttribute("SeedPack") ~= nil or loc:GetAttribute("GoldSeed") == true or loc:GetAttribute("RainbowSeed") == true
            if has then
                local rare = isRareSeedPack(loc)
                local grabGold = Library.Flags["collectGold"] == true
                local grabRainbow = Library.Flags["collectRainbow"] == true
                local isGold = loc:GetAttribute("GoldSeed") == true
                local isRainbow = loc:GetAttribute("RainbowSeed") == true
                -- Skip logic: pack qualifies if it matches ANY active toggle.
                -- If both gold+rainbow toggles are on, only gold/rainbow packs pass (no plain).
                -- If rareSeedOnly is on and neither gold/rainbow is on, restrict to rare packs.
                local allowed = false
                if grabGold and isGold then allowed = true end
                if grabRainbow and isRainbow then allowed = true end
                if not (grabGold or grabRainbow) then
                    if Library.Flags["rareSeedOnly"] then allowed = rare else allowed = true end
                end
                if not allowed then continue end
                if Library.Flags["rareNotify"] and rare then notify("Rare Seed Spawned", packType(loc) or "Rare pack - grabbing!") end
                _grabbingPacks[loc] = true
                task.spawn(function() pcall(grabPackRobust, loc); _grabbingPacks[loc] = nil end)
                break
            end
        end
    end
end

        return HttpService:JSONDecode(raw)

local function serverHop(lowPop)
    local servers = fetchServers() local pick
    for _, s in ipairs(servers) do
        if s.id ~= game.JobId and s.playing and s.maxPlayers and s.playing < s.maxPlayers then
            if lowPop then if not pick or s.playing < pick.playing then pick = s end else pick = s break end
        end
    end
    if pick then notify("Server Hop", "Hopping to " .. pick.playing .. "/" .. pick.maxPlayers) pcall(function() TeleportService:TeleportToPlaceInstance(game.PlaceId, pick.id, client) end)
    else notify("Server Hop", "No suitable server found", "warning") end
end

-- fly system (with proper off-state)
local _flyBV, _flyBG
local function stopFly()
    if _flyBV then pcall(function() _flyBV:Destroy() end) _flyBV = nil end
    if _flyBG then pcall(function() _flyBG:Destroy() end) _flyBG = nil end
    local h = getHumanoid() if h then h.PlatformStand = false end
end
local function doFlySystem(dt)
    if not Library.Flags["freeFlight"] then if _flyBV then stopFly() end return end
    local c = client.Character if not c then return end
    local hrp = c:FindFirstChild("HumanoidRootPart") if not hrp then return end
    local cam = Workspace.CurrentCamera if not cam then return end
    local h = c:FindFirstChildOfClass("Humanoid")
    if not _flyBV then
        _flyBV = Instance.new("BodyVelocity") _flyBV.MaxForce = Vector3.new(1, 1, 1) * 9e9 _flyBV.Velocity = Vector3.zero _flyBV.Parent = hrp
        _flyBG = Instance.new("BodyGyro") _flyBG.MaxTorque = Vector3.new(1, 1, 1) * 9e9 _flyBG.P = 1e5 _flyBG.CFrame = hrp.CFrame _flyBG.Parent = hrp
    end
    if h then h.PlatformStand = true end
    local speed = tonumber(Library.Flags["flightSpeed"]) or 50
    local dir = Vector3.zero
    if UserInputService:IsKeyDown(Enum.KeyCode.W) then dir = dir + cam.CFrame.LookVector end
    if UserInputService:IsKeyDown(Enum.KeyCode.S) then dir = dir - cam.CFrame.LookVector end
    if UserInputService:IsKeyDown(Enum.KeyCode.A) then dir = dir - cam.CFrame.RightVector end
    if UserInputService:IsKeyDown(Enum.KeyCode.D) then dir = dir + cam.CFrame.RightVector end
    if UserInputService:IsKeyDown(Enum.KeyCode.Space) then dir = dir + Vector3.new(0, 1, 0) end
    if UserInputService:IsKeyDown(Enum.KeyCode.LeftShift) then dir = dir - Vector3.new(0, 1, 0) end
    if _flyBV then _flyBV.Velocity = (dir.Magnitude > 0 and dir.Unit or Vector3.zero) * speed end
    if _flyBG then _flyBG.CFrame = cam.CFrame end
end

-- sell on interval (single driver — no more double-mechanism)
local _lastSell = 0
RunService.Heartbeat:Connect(function()
    local now = os.clock()
    if Library.Flags["autoSell"] then
        local si = tonumber(Library.Flags["sellInterval"]) or 20
        if now - _lastSell >= si then _lastSell = now doSellAll() end
    end
    -- sell on full
    if Library.Flags["sellOnFull"] and isInventoryFull() then doSellAll() end
    -- daily deal
    if Library.Flags["dailyDeal"] then netFire("NPCS.UseDailyDealAll") end
    -- bargain sell
    if Library.Flags["bargainSell"] then netFire("NPCS.SellAllBargain") end
end)

-- consolidated 2s loop: defense, notify, ESP, highlights, passives
task.spawn(function()
    local espTimer = 0
    local promptTimer = 0
    while task.wait(2) do
        pcall(function()
            if Library.Flags["hitStolen"] or Library.Flags["petProtect"] then doRetaliateShovel() end
            if Library.Flags["highlightReady"] or Library.Flags["highlightRare"] then doHighlightLoop() end
            if Library.Flags["rareNotify"] then doRareNotify() end
            if Library.Flags["antiFling"] then doAntiFling() end
            if Library.Flags["bypassPause"] then doBypassPause() end
            if Library.Flags["noclipPlants"] then doNoclipPlants() end
            if Library.Flags["lessKnockback"] then doLessKnockback() end
            -- auto sell selective / pets
            if Library.Flags["autoSellSelective"] then doSellSelective() end
            if Library.Flags["autoSellPets"] then doSellPets() end
            -- auto buy all
            if Library.Flags["autoBuyAllSeeds"] then local it = seedStock() if it then for _, sv in ipairs(it:GetChildren()) do if sv:IsA("ValueBase") and sv.Value > 0 and getBalance() >= (SeedPrice[sv.Name] or 0) then netFire("SeedShop.PurchaseSeed", sv.Name) task.wait(0.08) end end end end
            if Library.Flags["autoBuyAllGear"] then local it = gearStock() if it then for _, sv in ipairs(it:GetChildren()) do if sv:IsA("ValueBase") and sv.Value > 0 then netFire("GearShop.PurchaseGear", sv.Name) task.wait(0.08) end end end end
            if Library.Flags["autoBuyAllCrates"] then local it = crateStock() if it then for _, sv in ipairs(it:GetChildren()) do if sv:IsA("ValueBase") and sv.Value > 0 then netFire("CrateShop.PurchaseCrate", sv.Name) task.wait(0.08) end end end end
            -- autoSprinklerAll now handled inside doSprinkler when the flag is on
            -- (kept as an explicit driver so it works even when 'Auto Place Sprinkler' interval is off)
            if Library.Flags["autoSprinklerAll"] and not Library.Flags["autoSprinkler"] then doSprinkler() end
            -- autoPlantAll: driver when the user only toggled the "All Seeds" shortcut
            if Library.Flags["autoPlantAll"] and not Library.Flags["autoPlant"] then doPlant() end
            -- auto water all
            if Library.Flags["autoWaterAll"] then doWateringCan() end
            -- auto buy pet (experimental)
            if Library.Flags["autoBuyPet"] then if not _warnedBuyPet then _warnedBuyPet = true warn("[GAG2] Auto Buy Pet: experimental - remote not verified") end end
            -- auto favorite/unfavorite
            if Library.Flags["autoFav"] then doFavorite(true) end
            if Library.Flags["autoUnfav"] then doFavorite(false) end
            if Library.Flags["autoUnfavAll"] then doFavorite(false) end
            -- auto mailbox
            if Library.Flags["autoSendSeed"] or Library.Flags["autoSendPet"] then doMailboxSend() end
            -- ESP
            if Library.Flags["espFruitEnabled"] or Library.Flags["espPetEnabled"] then espTimer = espTimer + 1 if espTimer >= 1 then espTimer = 0 doESP() end
            elseif next(_espPool) then clearESP() end
            -- instant prompt
            if Library.Flags["instantPrompt"] then promptTimer = promptTimer + 1 if promptTimer >= 3 then promptTimer = 0 doInstantPrompt() end end
            -- auto remove gardens
            if Library.Flags["autoRemoveGardens"] then removeOtherGardens() end
                -- stats update
                local bal = getBalance() local elapsed = os.clock() - sessionStart
                local earned = bal - (sessionStartBal or bal) if earned > sessionEarned then sessionEarned = earned end
                
                -- Rolling 60s profit window
                local now = os.clock()
                table.insert(profitWindow, { t = now, s = bal })
                while #profitWindow > 1 and (now - profitWindow[1].t) > 60 do table.remove(profitWindow, 1) end
                local perMin = 0
                if #profitWindow > 1 then
                    local dt = now - profitWindow[1].t
                    if dt > 2 then perMin = (bal - profitWindow[1].s) / dt * 60 end
                end

                local fc = getFruitCount() local mc = getMaxCapacity()
            local function ul(tag, text) local lb = Misc:FindFirstChild(tag) if lb and lb.updateText then lb:updateText(text) end end
            ul("statBalance", "Balance: " .. fmtCash(bal))
            ul("statPerMin", "Per Minute: " .. fmtCash(perMin))
            ul("statSession", "Session Earned: " .. fmtCash(sessionEarned))
            local invWorth = select(1, inventoryValue())
            ul("statInvWorth", "Inventory Worth: " .. fmtCash(invWorth))
            ul("statFruitCount", "Fruit Count: " .. fc .. "/" .. mc)
            ul("statHarvested", "Crops Harvested: " .. sessionHarvests)
            ul("statSessionTime", "Session Time: " .. math.floor(elapsed / 60) .. "m")
            -- Timers panel: shop restocks + weather
            local rs = nextRestock("SeedShop") ul("timerSeeds", "Seed Shop: " .. (rs and fmtClock(rs) or "--:--"))
            local rg = nextRestock("GearShop") ul("timerGears", "Gear Shop: " .. (rg and fmtClock(rg) or "--:--"))
            local rc = nextRestock("CrateShop") ul("timerCrates", "Crate Shop: " .. (rc and fmtClock(rc) or "--:--"))
            local wx, ph, pd = currentWeather()
            ul("timerWeather", "Weather: " .. tostring(wx or "Clear"))
            ul("timerPhase", "Phase: " .. tostring(ph or "--") .. " (" .. fmtClock(pd or 0) .. ")")
        end)
    end
end)

    while task.wait(0.6) do
        pcall(function()
            if Library.Flags["autoCollect"] or Library.Flags["autoCollectAll"] or Library.Flags["autoCollectBest"] or Library.Flags["collectGold"] or Library.Flags["collectRainbow"] then
                if Library.Flags["autoCollectAll"] then doHarvest(false) elseif Library.Flags["autoCollectBest"] then doHarvest(true) end
                doPackGrab()
            end
            if Library.Flags["collectDropped"] then doCollectDropped() end
        end)
    end
end)

        -- pack return & panic harvest
        task.spawn(function()
            local wasNight = false
            while task.wait(1) do
                pcall(function()
                    local n = isNight()
                    if not wasNight and n and Library.Flags["panicHarvest"] then
                        notify("Defense", "Panic Harvesting! Night has fallen.", "warning")
                        doHarvest(false) -- harvest everything quickly
                    end
                    if wasNight and not n and (Library.Flags["autoCollect"] or Library.Flags["collectGold"] or Library.Flags["collectRainbow"]) then
                        local plot = myPlot() local sp = plot and plot:FindFirstChild("SpawnPoint")
                        if sp then teleport(sp.Position) notify("Event", "Event over - returned to garden") end
                    end
                    wasNight = n
                end)
            end
        end)

-- multi-jump (fixed precedence)
local jumped = 0
UserInputService.JumpRequest:Connect(function()
    if not Library.Flags["multiJump"] then return end
    local c = client.Character if not c then return end
    local h = c:FindFirstChildOfClass("Humanoid") if not h then return end
    if h.FloorMaterial == Enum.Material.Air then
        jumped = jumped + 1
        if jumped <= 3 then
            local hrp = c:FindFirstChild("HumanoidRootPart")
            if hrp then hrp.Velocity = Vector3.new(hrp.Velocity.X, (tonumber(Library.Flags["jumpHeight"]) or 7.2) * 50, hrp.Velocity.Z) end
        end
    else jumped = 0 end
end)

-- harvest tracking
task.spawn(function()
    local last = getFruitCount()
    while task.wait(1) do
        pcall(function()
            local cur = getFruitCount()
            if cur > last then sessionHarvests = sessionHarvests + (cur - last) end
            last = cur
        end)
    end
end)
