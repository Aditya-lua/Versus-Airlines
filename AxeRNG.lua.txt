function interval(tag, flag, delayTime, callback)
    if not Library then return end
    Library:CleanupConnectionsByTag(tag)
    delayTime = math.max(tonumber(delayTime) or 0.1, 0.05)
    if not Library.Flags or not Library.Flags[flag] then
        return
    end

    local last = 0
    local running = false
    local conn = RunService.Heartbeat:Connect(function()
        if not Library.Flags or not Library.Flags[flag] then
            Library:CleanupConnectionsByTag(tag)
            return
        end

        local current = os.clock()
        if running or current - last < delayTime then
            return
        end

        last = current
        running = true

        local spawnFn = task and task.spawn or spawn
        spawnFn(function()
            local startedAt = os.clock()
            local ok, err = pcall(callback)
            local elapsed = os.clock() - startedAt

            if not ok then
                warn("[interval:" .. tostring(tag) .. "]", err)
            elseif elapsed > 10 and os.clock() - (tag .. "_slowWarn"):len() < 0 then
                warn(string.format("[AxeRNG] slow interval %s took %.3fs", tostring(tag), elapsed))
            end

            local waitFn = task and task.wait or wait
            waitFn()
            running = false
        end)
    end)

    Library:TrackConnection(conn, tag)
end

end

function prettyPrint(data, indent)
    indent = indent or 0
    local prefix = string.rep("    ", indent)
    if type(data) ~= "table" then
        print(prefix .. tostring(data))
        return
    end
    for k, v in pairs(data) do
        if type(v) == "table" then
            print(prefix .. tostring(k) .. " = {")
            prettyPrint(v, indent + 1)
            print(prefix .. "}")
        else
            print(prefix .. tostring(k) .. " = " .. tostring(v))
        end
    end
end

function addLog(msg)
    print("[AxeRNG] " .. os.date("[%H:%M:%S] ") .. tostring(msg))
end

function formatNumber(n)
    if type(n) ~= "number" then return "0" end
    if n >= 1e15 then return string.format("%.2fQ", n / 1e15)
    elseif n >= 1e12 then return string.format("%.2fT", n / 1e12)
    elseif n >= 1e9 then return string.format("%.2fB", n / 1e9)
    elseif n >= 1e6 then return string.format("%.2fM", n / 1e6)
    elseif n >= 1e3 then return string.format("%.1fK", n / 1e3)
    else return tostring(math.floor(n)) end
end

function formatTime(s)
    if s < 60 then return math.floor(s) .. "s"
    elseif s < 3600 then return math.floor(s / 60) .. "m " .. math.floor(s % 60) .. "s"
    else return math.floor(s / 3600) .. "h " .. math.floor((s % 3600) / 60) .. "m" end
end

function safeRequire(module, label)
    if not module then
        warn("[AxeRNG] Missing module:", tostring(label or module))
        return nil
    end

    local ok, result = pcall(require, module)
    if not ok then
        warn("[AxeRNG] Failed to require", tostring(label or module), result)
        return nil
    end

    return result
end

function findPath(root, path)
    local node = root
    for _, name in ipairs(path) do
        if not node then return nil end
        node = node:FindFirstChild(name)
    end
    return node
end

function safeRequirePath(root, path, label)
    return safeRequire(findPath(root, path), label or table.concat(path, "."))
end

function findDeep(parent, name)
    if not parent then return nil end
    for _, child in ipairs(parent:GetChildren()) do
        if child.Name == name then return child end
        local found = findDeep(child, name)
        if found then return found end
    end
    return nil
end

RemotesFolder = findDeep(ReplicatedStorage, "_remotes") or ReplicatedStorage:FindFirstChild("_remotes", true)

function getRemote(name)
    if not RemotesFolder then return nil, nil end
    local folder = RemotesFolder:FindFirstChild(name)
    if not folder then return nil, nil end
    return folder:FindFirstChildWhichIsA("RemoteFunction", true), folder:FindFirstChildWhichIsA("RemoteEvent", true)
end

function remoteFetch(name, method, ...)
    local rf = getRemote(name)
    if not rf then return nil end
    local args = { ... }
    local ok, result = pcall(function() return rf:InvokeServer(method, unpackArgs(args)) end)
    if ok then return result end
    return nil
end

function getRarityDenom(axeId)
    local axe = DataCache.axes[axeId]
    return axe and axe.weight and math.max(1, math.round(1 / axe.weight)) or 0

function isZoneUnlocked(zoneId)
    if not ZoneData or not ZoneData.ZONES then return false end
    local zone = ZoneData.ZONES[zoneId]
    if not zone then return false end
    if not zone.cost then return true end
    local zones = getZones()
    return type(zones) == "table" and zones[zoneId] == true
end

function getNextUnlockableZone()
    if not ZoneData or not ZoneData.ZONES or not ZoneData.ZONE_ORDER then return nil end
    for _, zoneId in ipairs(ZoneData.ZONE_ORDER) do
        local zone = ZoneData.ZONES[zoneId]
        if zone and zone.cost and zone.cost.logs and not isZoneUnlocked(zoneId) then
            if not zone.requires or isZoneUnlocked(zone.requires) then
                if getLogs() >= zone.cost.logs then
                    return zoneId
                end
            end
        end
    end
    return nil
end

function getBestAffordableZone()
    if not ZoneData or not ZoneData.ZONES or not ZoneData.ZONE_ORDER then return nil end
    for i = #ZoneData.ZONE_ORDER, 1, -1 do
        local zoneId = ZoneData.ZONE_ORDER[i]
        if isZoneUnlocked(zoneId) then
            return zoneId
        end
    end
    return nil
end

function getRebirthHoneyCost()
    local r = getRebirths()
    if RebirthState and RebirthState.honeyCost then
        local ok, cost = pcall(function() return RebirthState.honeyCost(r) end)
        if ok and cost then return cost end
    end
    return math.floor(1000 * 2^math.max(0, r) + 0.5)
end

function isSkillNodePurchased(nodeId)
    local st = getTableStat("skillTree")
    return type(st) == "table" and st[nodeId] ~= nil
end

function getPart(inst)
    if not inst or typeof(inst) ~= "Instance" then return nil end
    if inst:IsA("BasePart") then return inst end
    if inst:IsA("Model") then
        return inst.PrimaryPart or inst:FindFirstChildWhichIsA("BasePart", true)
    end
    if inst.FindFirstChildWhichIsA then
        return inst:FindFirstChildWhichIsA("BasePart", true)
    end
    return nil
end

function moveTo(target)
    local hrp = getHRP()
    local hum = getHumanoid()
    if not hrp or not hum then return end
    local dist = (target - hrp.Position).magnitude
    if dist < 3 then return end
    local moveMode = (Library and Library.Flags and Library.Flags["MoveMode"]) or "Walk"
    if moveMode == "TP" then
        hrp.CFrame = CFrame.new(target)
        return
    end
    local timeLimit = math.min(dist / 16, 8)
    hum:MoveTo(target)
    local start = os.clock()
    while os.clock() - start < timeLimit do
        local r = getHRP()
        if not r then break end
        if (target - r.Position).magnitude < 5 then break end
        if moveMode == "Tween" then
            r.CFrame = r.CFrame:Lerp(CFrame.new(target), 0.1)
        end
        task.wait()
    end
end

function teleportNear(pos, yOffset)
    local hrp = getHRP()
    if not hrp then return false end
    hrp.CFrame = CFrame.new(pos + Vector3.new(0, yOffset or 4, 0))
    return true
end

FarmZoneId = nil
FarmCenter = nil
CurrentTree = nil

function getTreeZone(treeModel)
    if not treeModel then return nil end
    if ZoneData and ZoneData.zoneForInstance then
        local ok, zoneId = pcall(function() return ZoneData.zoneForInstance(treeModel) end)
        if ok and zoneId then return zoneId end
    end
    local attr = treeModel:GetAttribute("Zone") or treeModel:GetAttribute("zoneId") or treeModel:GetAttribute("zone")
    if attr then return tostring(attr) end
    for _, tag in ipairs(CollectionService:GetTags(treeModel)) do
        local zid = tag:match("^zone:(.+)$")
        if zid then return zid end
    end
    return nil
end

function getCurrentZone()
    local hrp = getHRP()
    if not hrp then return nil end
    return getTreeZone(hrp)
end

function ensureFarmCenter()
    if not FarmCenter then
        local hrp = getHRP()
        if hrp then FarmCenter = hrp.Position end
    end
    if not FarmZoneId then
        local useBestZone = Library and Library.Flags and Library.Flags["FarmInBestZone"]
        if useBestZone then
            FarmZoneId = getBestAffordableZone()
        else
            FarmZoneId = getCurrentZone()
        end
    end
end

function doChop(tree)
    if not tree or not tree.chopId or not ChopTreeRE then return end
    pcall(function() firesignal(ChopTreeRE.OnClientEvent, "d", tree.chopId) end)
    pcall(function()
        if LogDropRE then
            local amt = math.random(1, 3)
            firesignal(LogDropRE.OnClientEvent, "ls", tree.chopId, "Log", amt)
        end
    end)
end

function scanFarmTrees()
    local hrp = getHRP()
    if not hrp then return {} end
    ensureFarmCenter()
    local trees = {}
    for _, obj in ipairs(CollectionService:GetTagged("Choppable")) do
        local rootModel = ChopNet and ChopNet.resolveChoppableRoot and ChopNet.resolveChoppableRoot(obj) or obj:FindFirstAncestorWhichIsA("Model")
        local chopId = ChopNet and ChopNet.getChopNetIdFromInstance and ChopNet.getChopNetIdFromInstance(obj) or obj:GetAttribute("ChopNetId")
        if rootModel and chopId and rootModel.PrimaryPart and not DepletedTrees[chopId] then
            local treePos = rootModel.PrimaryPart.Position
            local dist = (treePos - hrp.Position).magnitude
            if dist <= 300 then
                local treeZone = getTreeZone(rootModel)
                local zoneOk = not FarmZoneId or not treeZone or treeZone == FarmZoneId
                local leashOk = not FarmCenter or (treePos - FarmCenter).magnitude <= 300
                if zoneOk and leashOk then
                    table.insert(trees, { model = rootModel, chopId = chopId, pos = treePos, dist = dist })
                end
            end
        end
    end
    table.sort(trees, function(a, b) return a.dist < b.dist end)
    return trees
end

function guessDropId(part)
    if not part then return "Log" end
    if part.Size and part.Size.X >= 4 then return "Honeycomb" end
    if DropData and DropData.DROPS then
        local bb = part:FindFirstChildWhichIsA("BillboardGui")
        if bb then
            local icon = bb:FindFirstChild("DropIcon")
            if icon and icon:IsA("ImageLabel") then
                local img = icon.Image
                for did, dd in pairs(DropData.DROPS) do
                    if dd.imageKey and img:find(dd.imageKey, 1, true) then return did end
                end
            end
        end
    end
    return "Log"
end

function scanDrops()
    local hrp = getHRP()
    if not hrp then return {} end
    local drops = {}
    for _, part in ipairs(CollectionService:GetTagged("LogDrop")) do
        local dist = (part.Position - hrp.Position).magnitude
        if dist <= 30 then
            local dropId = guessDropId(part)
            local count = 1
            local bb = part:FindFirstChildWhichIsA("BillboardGui")
            if bb then
                local label = bb:FindFirstChildWhichIsA("TextLabel")
                if label then count = tonumber(label.Text:match("x(%d+)")) or 1 end
            end
            table.insert(drops, { part = part, dropId = dropId, count = count, pos = part.Position, dist = dist })
        end
    end
    table.sort(drops, function(a, b) return a.dist < b.dist end)
    return drops
end

function tryChop(tree)
    if not tree or not tree.chopId then return end
    doChop(tree)
    DepletedTrees[tree.chopId] = true
    if tree.model then
        local cd = tree.model:FindFirstChildWhichIsA("ClickDetector", true)
        if cd then pcall(function() fireclickdetector(cd) end); return end
        local pp = tree.model:FindFirstChildWhichIsA("ProximityPrompt", true)
        if pp then pcall(function() fireproximityprompt(pp) end); return end
    end
end

function tryCollectDrop(drop)
    if not drop then return end
    if LogDropNetwork and LogDropNetwork.collectDrop then
        pcall(function() LogDropNetwork.collectDrop(drop.dropId, drop.count) end)
    else
        pcall(function() remoteFetch("LogDrop", "collectDrop", drop.dropId, drop.count) end)
    end
end

function stepFarm()

        CurrentTree = target

    local zoneId = getBestAffordableZone()
    if zoneId then
        pcall(function() remoteFetch("Zone", "teleportToZone", zoneId) end)
        task.wait(0.5)
        FarmZoneId = zoneId
        local hrp = getHRP()
        if hrp then FarmCenter = hrp.Position end
        addLog("TP'd to best zone: " .. zoneId)
    end
end

function stepAntiAfk()
    local hum = getHumanoid()
    local hrp = getHRP()
    if hum and hrp then hum:MoveTo(hrp.Position) end
    pcall(function() VirtualUser:CaptureController() end)
end

function redeemCodes()
    for _, code in ipairs(DataCache.codes) do
        if not RedeemedCodes[code] then
            RedeemedCodes[code] = true
            pcall(function() remoteFetch("Codes", "redeem", code) end)
            addLog("Redeemed: " .. code)
            task.wait(0.3)
        end
    end
end

AllToggles = {}

function createAxeToggle(section, cfg)
    table.insert(AllToggles, cfg)
    local tag = cfg.tag or ("AxeRNG_" .. cfg.flagName)
    section:createToggle({
        Name = cfg.Name,
        Flag = cfg.Flag or false,
        flagName = cfg.flagName,
        Callback = function(enabled)
            Library:CleanupConnectionsByTag(tag)
            if not enabled then return end
            interval(tag, cfg.flagName, cfg.delay or 1, cfg.Step)
        end,
    })
end

----------------------------------------------------------------------
-- Stats updater (always-on interval)
----------------------------------------------------------------------

local fpsValue = 0
local statsConn = RunService.Heartbeat:Connect(function(dt)
    fpsValue = math.floor(1 / dt)
end)
Library:TrackConnection(statsConn, "AxeRNG_StatsFPS")

interval("AxeRNG_StatsUpdate", "AxeRNG_StatsUpdate", 1, function()
    local logs = getLogs()
    local honey = getHoney()
    local crystals = getCrystals()
    local rebirths = getRebirths()
    local uptime = os.clock() - Stats.startTime

    Sl.logs:Set("Logs: " .. formatNumber(logs))
    Sl.honey:Set("Honey: " .. formatNumber(honey))
    Sl.crystals:Set("Crystals: " .. formatNumber(crystals))
    Sl.rebirths:Set("Rebirths: " .. formatNumber(rebirths))
    Sl.uptime:Set("Uptime: " .. formatTime(uptime))
    Sl.chops:Set("Chops: " .. formatNumber(Stats.totalChops))
    Sl.rolls:Set("Rolls: " .. Stats.totalRolls)

    if Stats.bestRollAxe then
        local a = DataCache.axes[Stats.bestRollAxe]
        Sl.bestRoll:Set("Best: " .. (a and a.name or Stats.bestRollAxe) .. " (1 in " .. Stats.bestRollRarity .. ")")
    end

    Sl.fps:Set("FPS: " .. fpsValue)
end)

Library.Flags["AxeRNG_StatsUpdate"] = true
